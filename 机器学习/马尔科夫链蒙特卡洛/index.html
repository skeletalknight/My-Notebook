def reject_or_accept(theta_cur, theta_cand, pdf):
    return min(1, pdf(theta_cur) / pdf(theta_cand))  # ËøîÂõûÊé•ÂèóÊ¶ÇÁéá

theta_cur = 0.5  # ÂàùÂßãÂÄº
std_dev = 0.1  # Ê†áÂáÜÂ∑Æ

n_samples = 100000  # ÈááÊ†∑Ê¨°Êï∞
sample_array = []

while len(sample_array) < n_samples:
    theta_cand = np.random.normal(theta_cur, std_dev)  # ÁîüÊàêÂÄôÈÄâÂÄº
    if reject_or_accept(theta_cur, theta_cand) > np.random.rand():
        theta_cur = theta_cand  # Êé•ÂèóÂÄôÈÄâÂÄº
    if burn_in > 0:
        burn_in -= 1

E_theta = np.mean(sample_array)  # ËÆ°ÁÆóÊúüÊúõÂÄº
Var_theta = np.var(sample_array)  # ËÆ°ÁÆóÊñπÂ∑Æ

print("ÊñπÂ∑Æ Var[Œ∏]:", Var_theta)
